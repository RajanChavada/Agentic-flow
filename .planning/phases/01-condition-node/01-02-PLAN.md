   ```typescript
   // Condition edge coloring: green for True branch, red for False branch
   // Only applies when no external override (e.g., critical path blue) is present
   const defaultStrokeColor = "#6b7280";
   const isDefaultStroke = !style?.stroke || style.stroke === defaultStrokeColor;

   let resolvedStroke: string | undefined;
   if (isDefaultStroke && sourceHandleId?.includes("true")) {
     resolvedStroke = "#22c55e"; // green-500 for True branch
   } else if (isDefaultStroke && sourceHandleId?.includes("false")) {
     resolvedStroke = "#ef4444"; // red-500 for False branch
   }

   const finalStyle = resolvedStroke
     ? { ...style, stroke: resolvedStroke, strokeWidth: 2 }
     : style;

   const finalMarkerEnd = resolvedStroke
     ? { type: MarkerType.ArrowClosed, width: 16, height: 16, color: resolvedStroke }
     : markerEnd;
   ```
